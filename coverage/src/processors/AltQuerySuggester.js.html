      {
        const
        {
          keyword,
          hits

        if (hits)

  getClosest(word)
  {
    let closest, closestDiff = 100;
    const dist = levenshtein.get.bind(levenshtein);
    for (const target of Object.keys(this.words))
    {
      const diff = dist(word, target) / Math.min(word.length, target.length);
      if (diff < closestDiff)

        closestDiff = diff;
        closest = target;
      }
    }
