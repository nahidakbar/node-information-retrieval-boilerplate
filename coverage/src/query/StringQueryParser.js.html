  * tokenise(string)
  {
    for (let [char, class_] of this.lemmatise(string))
    {
      switch (class_)
      {
      case 'a':
        switch (char.toLowerCase())

        case 'and':
          yield [char, 'and'];
          continue;
        case 'or':
          yield [char, 'or'];
          continue;
        case 'not':
          yield [char, 'not'];
          continue;

        yield [char, 'not'];
        continue;
      case ' ':

      case '0':

        continue;

  * lex(string)
  {

  classifyChar(char)
  {
    switch (char)
    {
      // whitespace
    case '"':
    case "'":
    case "(":
    case ")":
    case "[":
    case "]":
      return '"';
    case ':':
      return ':';
    case '-':
      return '-';
    default:
      if ((char >= 'A' && char <= 'Z') || (char >= 'a' && char <= 'z'))
      {
        return 'a'
      }
      else if ((char >= '0' && char <= '9') || (char === '.'))
      {
        return '0'

      else
      {
